ZJ b557: 直角三角形

 1#include <bits/stdc++.h>
 2using namespace std;
 3vector<tuple<int, int, int>> v;
 4void init() {
 5    for (int i = 1; i <= 100; i++) {
 6        for (int j = i; j <= 100; j++) {
 7            for (int k = j; k <= 100; k++) {
 8                if (i * i + j * j == k * k) {
 9                    v.emplace_back(i, j, k);
10                }
11            }
12        }
13    }
14}
15int main() {
16    init();
17    int T;
18    cin >> T;
19    while (T--) {
20        int n;
21        cin >> n;
22        int cnt[101] = {};
23        for (int i = 0; i < n; i++) {
24            int l;
25            cin >> l;
26            cnt[l]++;
27        }
28        long long ans = 0;
29        for (auto t : v) {
30            ans += cnt[get<0>(t)] * cnt[get<1>(t)] * cnt[get<2>(t)];

CSES1707 - Graph Girth

  • 枚舉拔每條邊 (i, j)，做 BFS 找 i, j 的最短路
 1#pragma GCC optimization ("O3")
 2#include<bits/stdc++.h>
 3using namespace std;
 4typedef long long ll;
 5typedef pair<int, int> pii;
 6#define pb push_back
 7
 8int main() {
 9    ios::sync_with_stdio(0), cin.tie(0);
10    int n, m;
11    cin >> n >> m;
12    int INF = n + 10;
13    vector<vector<int>> G(n + 1);
14    for (int i = 0, a, b; i < m; i++) {
15        cin >> a >> b;
16        G[a].pb(b);
17        G[b].pb(a);
18    }
19    int ans = INF;
20    for (int i = 1; i <= n; i++) {
21        for (int j : G[i]) {
22            vector<bool> visit(n + 1);
23            queue<int> q, dis;
24            q.push(i), dis.push(1);
25            visit[i] = true;
26            while (!q.empty()) {
27                int u = q.front();
28                q.pop();
29                int d = dis.front();
30                dis.pop();
31                if (u == j) {
32                    ans = min(ans, d);
33                    break;
34                }
35                for (int uu : G[u]) {
36                    if ((i == u && j == uu) || (j == u && i == uu))
37                        continue;
38                    if (!visit[uu]) {
39                        visit[uu] = true;
40                        q.push(uu), dis.push(d + 1);
41                    }
42                }
43            }
44        }
45    }
46    if (ans == INF) ans = -1;
47        cout << ans << "\n";
48    return 0;
49}
