Append Sort

 1#pragma GCC optimization ("O3")
 2#include<bits/stdc++.h>
 3using namespace std;
 4typedef long long ll;
 5string int_to_string(int i) {
 6  string s = "";
 7  while (i) {
 8    s += char('0' + i % 10);
 9    i /= 10;
10  }
11  reverse(s.begin(), s.end());
12  return s;
13}
14int main() {
15  ios::sync_with_stdio(0), cin.tie(0);
16  int T; cin >> T;
17  for (int ti = 1; ti <= T; ti++) {
18    int n; cin >> n;
19    vector<int> a(n);
20    for (int i = 0; i < n; i++) cin >> a[i];
21    vector<string> s(n);
22    for (int i = 0; i < n; i++)
23      s[i] = int_to_string(a[i]);
24    int ans = 0;
25    for (int i = 1; i < n; i++) {
26      if (s[i - 1].length() < s[i].length()) continue;
27      if (s[i - 1].length() == s[i].length()) {
28        if (s[i] <= s[i - 1])
29          s[i].push_back('0'), ans++;
30        continue;
31      }
32      string sub = s[i - 1].substr(0, s[i].length());
33      if (sub > s[i]) {
34        int cnt = s[i - 1].length() - s[i].length() + 1;
35        ans += cnt;
36        while (cnt--) s[i].push_back('0');
37        continue;
38      }
39      if (sub == s[i]) {
40        string tmp = s[i - 1];
41        for (int j = s[i - 1].length() - 1; j >= s[i].length(); j--) {
42          if (s[i - 1][j] != '9') {
43            tmp[j]++;
44            for (int k = j + 1; k < s[i - 1].length(); k++)
45              tmp[k] = '0';
46            break;
47          }
48        }
49        if (tmp != s[i - 1]) {
50          ans += s[i - 1].length() - s[i].length(), s[i] = tmp;
51        } else {
52          int cnt = s[i - 1].length() - s[i].length() + 1;
53          ans += cnt;
54          while (cnt--) s[i].push_back('0');
55        }
56        continue;
57      }
58      int cnt = s[i - 1].length() - s[i].length();
59      ans += cnt;
60      while (cnt--) s[i].push_back('0');
61    }
62    cout << "Case #" << ti << ": " << ans << '\n';
63  }
64  return 0;
65}

Prime Time

 1#pragma GCC optimization ("O3")
 2#include<bits/stdc++.h>
 3using namespace std;
 4int main() {
 5    ios::sync_with_stdio(0), cin.tie(0);
 6    int T; cin >> T;
 7    for (int ti = 1; ti <= T; ti++) {
 8        int m; cin >> m;
 9        vector<int> n(m), p(m);
10        int max_sum = 0;
11        for (int i = 0; i < m; i++) {
12            cin >> p[i] >> n[i];
13            max_sum += p[i] * n[i];
14        }
15        int ans = 0;
16        for (int i = max_sum - 1; i > 1; i--) {
17            int sum = 0, tmp = i;
18            for (int j = 0; j < m; j++)
19                for (int k = 0; k < n[j]; k++)
20                    if (tmp % p[j]) sum += p[j];
21                    else tmp /= p[j];
22            if (tmp == 1 && sum == i) {
23                ans = i;
24                break;
25            }
26        }
27        cout << "Case #" << ti << ": " << ans << '\n';
28    }
29    return 0;
30}

Hacked Exam

 1#pragma GCC optimization ("O3")
 2#include<bits/stdc++.h>
 3using namespace std;
 4int main() {
 5    ios::sync_with_stdio(0), cin.tie(0);
 6    int T; cin >> T;
 7    for (int ti = 1; ti <= T; ti++) {
 8        int n, q; cin >> n >> q;
 9        vector<string> s(n);
10        vector<int> score(n);
11        for (int i = 0; i < n; i++) cin >> s[i] >> score[i];
12        int ans_num = 0;
13        vector<int> t(q);
14        for (int i = 0; i < (1 << q); i++) {
15            string tmp = "";
16            for (int j = 0; j < q; j++)
17                if ((1 << j) & i) tmp += 'T';
18                else tmp += 'F';
19            vector<int> cnt(n);
20            bool legal = true;
21            for (int j = 0; j < n; j++) {
22                for (int k = 0; k < q; k++)
23                    if (s[j][k] == tmp[k]) cnt[j]++;
24                if (cnt[j] != score[j]) {
25                    legal = false;
26                    break;
27                }
28            }
29            if (!legal) continue;
30            ans_num++;
31            for (int j = 0; j < q; j++)
32                if ((1 << j) & i) t[j]++;
33        }
34        cout << "Case #" << ti << ": ";
35        int cnt = 0;
36        for (int i = 0; i < q; i++) {
37            if (t[i] > ans_num - t[i]) cout << 'T';
38            else cout << 'F';
39            cnt += max(t[i], ans_num - t[i]);
40        }
41        int g = gcd(cnt, ans_num);
42        cout << ' ' << cnt / g << '/' << ans_num / g << '\n';
43    }
44    return 0;
45}
