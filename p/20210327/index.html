Problem - Reversort

 1#include<bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    ios::sync_with_stdio(0), cin.tie(0);
 5    int T; cin >> T;
 6    for (int ti = 1; ti <= T; ti++) {
 7        int n; cin >> n;
 8        vector<int> v(n);
 9        for (int i = 0; i < n; i++) cin >> v[i];
10        int cost = 0;
11        for (int i = 0, j; i < n - 1; i++) {
12            for (j = i; j < n; j++)
13                if (v[j] == i + 1) break;
14            cost += j - i + 1;
15            for (int k = i, r = j; k < r; k++, r--) {
16                int tmp = v[k]; v[k] = v[r]; v[r] = tmp;
17            }
18        }
19        cout << "Case #" << ti << ": " << cost << "\n";
20    }
21    return 0;
22}

Problem - Moons and Umbrellas

 1#include<bits/stdc++.h>
 2using namespace std;
 3#define INF 1e9
 4int main() {
 5    ios::sync_with_stdio(0), cin.tie(0);
 6    int T; cin >> T;
 7    for (int ti = 1; ti <= T; ti++) {
 8        int x, y;
 9        string s;
10        cin >> x >> y >> s;
11        int n = s.length();
12        vector<int> DPC(n, INF), DPJ(n, INF);
13        DPC[0] = s[0] == 'J' ? INF : 0;
14        DPJ[0] = s[0] == 'C' ? INF : 0;
15        for (int i = 1; s[i]; i++) {
16            if (s[i] == 'C') {
17                DPC[i] = min(DPC[i - 1], DPJ[i - 1] + y);
18                DPJ[i] = INF;
19            } else if (s[i] == 'J') {
20                DPJ[i] = min(DPJ[i - 1], DPC[i - 1] + x);
21                DPC[i] = INF;
22            } else {
23                DPJ[i] = min(DPJ[i - 1], DPC[i - 1] + x);
24                DPC[i] = min(DPC[i - 1], DPJ[i - 1] + y);
25            }
26        }
27        cout << "Case #" << ti << ": "
28             << min(DPC[n - 1], DPJ[n - 1]) << "\n";
29    }
30    return 0;
31}
