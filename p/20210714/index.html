 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    int n, m;
 5    cin >> n >> m;
 6    vector<vector<int>> G(n + 1);
 7    for (int i = 0; i < m; ++i) {
 8        int u, v;
 9        cin >> u >> v;
10        G[u].push_back(v);
11        G[v].push_back(u);
12    }
13    if (m != n - 1) {
14        printf("NO\n");
15    } else {
16        vector<bool> visit(n + 1);
17        queue<int> q;
18        q.push(1);
19        visit[1] = true;
20        while (!q.empty()) {
21            int u = q.front();
22            q.pop();
23            for (int uu : G[u]) {
24                if (!visit[uu]) {
25                    q.push(uu);
26                    visit[uu] = true;
27                }
28            }
29        }
30        for (int i = 1; i <= n; ++i) {
31            if (!visit[i]) {
32                printf("NO\n");
33                return 0;
34            }
35        }
36        printf("YES\n");
37    }
38    return 0;
39}

 1#include <bits/stdc++.h>
 2using namespace std;
 3#define maxn 100005
 4#define maxl 80
 5long long dp[maxn][maxl] = {};
 6int main() {
 7  ios::sync_with_stdio(0);
 8  cin.tie(0);
 9  long long n, r, b;
10  cin >> n >> r >> b;
11  dp[0][0] = 1;
12  for (int i = 1; i < maxn; i++) {
13    for (int j = 0; j < maxl; j++) {
14      dp[i][j] += dp[i - 1][j] * r;
15      dp[i][j + 1] += dp[i][j] / 10;
16      dp[i][j] %= 10;
17    }
18    dp[i][0] += b;
19    for (int j = 0; j < maxl; j++) {
20      while (dp[i][j] < 0) {
21        dp[i][j + 1]--;
22        dp[i][j] += 10;
23      }
24    }
25  }
26  while (n--) {
27    int x;
28    cin >> x;
29    string ans = "";
30    bool flag = false;
31    for (int i = maxl - 1; i >= 0; i--) {
32      if (dp[x][i])
33        flag = true;
34      if (!flag)
35        continue;
36      ans += char(dp[x][i] + '0');
37    }
38    if (ans.length() == 0)
39      ans = "0";
40    cout << ans << "\n";
41  }
42  return 0;
43}

 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4  ios::sync_with_stdio(0);
 5  cin.tie(0);
 6  int n;
 7  cin >> n;
 8  vector<int> a(n), c(n);
 9  int sum = 0;
10  for (int i = 0; i < n; i++) {
11    cin >> a[i] >> c[i];
12    sum += a[i];
13  }
14  vector<int> dp(sum + 1, n * 1000);
15  dp[0] = 0;
16  for (int i = 0; i < n; i++) {
17    for (int j = sum; j >= a[i]; j--)
18      dp[j] = min(dp[j], dp[j - a[i]] + c[i]);
19  }
20  for (int j = sum - 1; j >= 0; j--)
21    dp[j] = min(dp[j + 1], dp[j]);
22  int T;
23  cin >> T;
24  while (T--) {
25    long long P;
26    cin >> P;
27    if (P > sum) {
28      cout << 0 << "\n";
29    } else {
30      cout << dp[P] << "\n";
31    }
32  }
33  return 0;
34}

 1#include <bits/stdc++.h>
 2using namespace std;
 3typedef pair<int, int> pii;
 4int main() {
 5  ios::sync_with_stdio(0);
 6  cin.tie(0);
 7  int N, M, P;
 8  cin >> N >> M >> P;
 9  queue<int> qx, qy;
10  vector<vector<int>> dis(N, vector<int>(M, -1));
11  for (int i = 0; i < P; i++) {
12    int x, y;
13    cin >> x >> y;
14    qx.push(x);
15    qy.push(y);
16    dis[x][y] = 0;
17  }
18  int ans = 0;
19  int dx[4] = {1, -1, 0, 0};
20  int dy[4] = {0, 0, 1, -1};
21  while (!qx.empty()) {
22    int x = qx.front();
23    qx.pop();
24    int y = qy.front();
25    qy.pop();
26    for (int di = 0; di < 4; di++) {
27      int xx = x + dx[di];
28      int yy = y + dy[di];
29      if (xx < 0 || xx >= N || yy < 0 || yy >= M)
30        continue;
31      if (dis[xx][yy] != -1) continue;
32      dis[xx][yy] = dis[x][y] + 1;
33      qx.push(xx), qy.push(yy);
34      ans = max(ans, dis[xx][yy]);
35    }
36  }
37  cout << ans << "\n";
38  return 0;
39}
