ZJ. f698: 後序運算式

 1#include <bits/stdc++.h>
 2using namespace std;
 3typedef long long ll;
 4int main() {
 5    ios::sync_with_stdio(0);
 6    cin.tie(0);
 7    string s;
 8    stringstream ss;
 9    getline(cin, s);
10    ss.str(s);
11    vector<ll> v;
12    while (ss >> s) {
13        if (isdigit(s.back())) {
14            v.push_back(stoi(s));
15        } else {
16            ll r = v.back(); v.pop_back();
17            ll l = v.back(); v.pop_back();
18            if (s[0] == '+') {
19                v.push_back(l + r);
20            } else if (s[0] == '-') {
21                v.push_back(l - r);
22            } else if (s[0] == '*') {
23                v.push_back(l * r);
24            } else {
25                v.push_back(l / r);
26            }
27        }
28    }
29    cout << v.back() << "\n";
30    return 0;
31}

 8        if (!n) break;
 9        if (!ans) ans = n;
10        if (n % 7 == 0 && ans % 7 != 0)
11            ans = n;
12        else if (ans % 7 == 0 && n % 7 == 0 && n % 70 > ans % 70)

14        else if (ans % 7 != 0 && n % 7 != 0 && n % 77 < ans % 77)

ZJ. f885: 加總

 1#include <bits/stdc++.h>
 2using namespace std;
 3typedef long long ll;
 4int main() {
 5    ios::sync_with_stdio(0);
 6    cin.tie(0), cout.tie(0);
 7    int T;
 8    cin >> T;
 9    while (T--) {
10        ll a, x;
11        cin >> a >> x;
12        ll L = a, R = x;
13        while (L <= R) {
14            ll mid = (L + R) >> 1;
15            ll sum = (a + mid) * (mid - a + 1) >> 1;
16            if (sum < x) L = mid + 1;
17            else R = mid - 1;
18        }
19        cout << L << "\n";
20    }
21    return 0;
22}

ZJ. f680: 色塊數量

BFS

 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    int n;
 5    cin >> n;
 6    vector<vector<int>> v(n, vector<int>(n));
 7    for (int i = 0; i < n; ++i)
 8        for (int j = 0; j < n; ++j)
 9            cin >> v[i][j];
10    const int dx[4] = {1, -1, 0, 0};
11    const int dy[4] = {0, 0, 1, -1};
12    int ans = 0;
13    for (int i = 0; i < n; ++i) {
14        for (int j = 0; j < n; ++j) {
15            if (v[i][j] == 0) continue;
16            ++ans;
17            int color = v[i][j];
18            queue<int> qx, qy;
19            qx.push(i), qy.push(j);
20            v[i][j] = 0;
21            while (!qx.empty()) {
22                int x = qx.front(); qx.pop();
23                int y = qy.front(); qy.pop();
24                for (int di = 0; di < 4; ++di) {
25                    int xx = x + dx[di];
26                    int yy = y + dy[di];
27                    if (xx < 0 || xx >= n) continue;
28                    if (yy < 0 || yy >= n) continue;
29                    if (v[xx][yy] == color) {
30                        qx.push(xx), qy.push(yy);
31                        v[xx][yy] = 0;
32                    }
33                }
34            }
35        }
36    }
37    cout << ans << '\n';
38    return 0;
39}

ZJ. f416: 果然我的期中程設考搞錯了什麼

rolling hash

 1#include <bits/stdc++.h>
 2using namespace std;
 3typedef long long ll;
 4const int P = 1000000007;
 5ll modinv(ll a, ll p) {
 6    ll pp = p, y = 0, x = 1;
 7    while (a > 1) {
 8        ll q = a / p,t = p;
 9        p = a % p, a = t, t = y, y = x - q * y, x = t;
10    }
11    if (x < 0) x += pp;
12    return x;
13}
14int main() {
15    ios::sync_with_stdio(0), cin.tie(0);
16    int n;
17    string a, b;
18    cin >> n >> a >> b;
19    ll hash = 0;
20    ll x = 1;
21    for (int i = 1; i < a.length(); ++i)
22        x = x * 26 % P;
23    for (int i = 0; a[i]; ++i) {
24        int c = a[i] - 'A';
25        hash = (hash * 26 + c) % P;
26    }
27    int len = a.length();
28    ll now = 0;
29    ll inv = modinv(26, P);
30    int cnt = 0;
31    for (int i = 0; b[i]; ++i) {
32        int c = b[i] - 'A';
33        if (i >= len)
34            now = (now - x * (b[i - len] - 'A') % P + P) % P;
35        now = (now * 26 + c) % P;
36        if (i >= len - 1 && now == hash) ++cnt;
37    }
38    cout << (cnt == n ? "YES" : "NO") << '\n';
39    return 0;
40}
