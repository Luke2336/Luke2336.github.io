EDA HW1

F-M Algorithm

  • 將 Bug 改掉，結果進步了不少。
  • 昨天的版本是每個 Partition 只做一次 F-M，但改成做多次直到 cut 數量收斂，結果進步了 3 倍以上。
  • 因為是平行化同步做，所以每個 Thread 要給不同的 Random Seed。

  1#include <bits/stdc++.h>
  2using namespace std;
  3class BucketList {
  4  map<int, unordered_set<int>> mp[2];
  5  unordered_map<int, pair<int, int>> Gain;
  6
  7public:
  8  int mx(int b) {
  9    while (mp[b].size() && mp[b].rbegin()->second.empty())
 10      mp[b].erase(mp[b].rbegin()->first);
 11    if (mp[b].size() == 0) return -1;
 12    return mp[b].rbegin()->first;
 13  }
 14  int pop(int b) {
 15    if (mx(b) < 0) return -1;
 16    int ret = *(mp[b].at(mx(b)).begin());
 17    mp[b].at(mx(b)).erase(ret);
 18    return ret;
 19  }
 20  void remove(int v) { mp[Gain.at(v).first].at(Gain.at(v).second).erase(v); }
 21  void add(int v, int b, int id) {
 22    Gain[v] = make_pair(b, id);
 23    mp[b][id].insert(v);
 24  }
 25};
 26inline bool range(int v, int L, int R) { return L <= v && v <= R; }
 27vector<vector<int>> Nets;
 28vector<vector<int>> Cells;
 29int FM(vector<bool> &Partition, const int P) {
 30  int N = Cells.size(), M = Nets.size();
 31  int M1 = ceil(N * 0.451), M2 = floor(N * 0.549);
 32  vector<bool> Lock(N);
 33  vector<int> Gain(N), CntCell[2];
 34  CntCell[0].resize(M), CntCell[1].resize(M);
 35  for (int i = 0; i < M; i++) {
 36    for (int j : Nets[i])
 37      CntCell[Partition[j]][i]++;
 38    for (int j : Nets[i])
 39      if (!CntCell[!Partition[j]][i]) --Gain[j];
 40      else if (CntCell[Partition[j]][i] == 1) ++Gain[j];
 41  }
 42  BucketList BL;
 43  for (int i = 0; i < N; i++)
 44    BL.add(i, Partition[i], Gain[i]);
 45  vector<int> sumGain;
 46  vector<int> swapNode;
 47  int sz[2] = {};
 48  for (auto i : Partition) sz[i]++;
 49  for (int t = 0; t < N; ++t) {
 50    bool b;
 51    if (range(sz[0] - 1, M1, M2) && BL.mx(0) >= BL.mx(1)) b = 0;
 52    else if (range(sz[0] + 1, M1, M2) && BL.mx(0) <= BL.mx(1)) b = 1;
 53    else if (range(sz[0] - 1, M1, M2)) b = 0;
 54    else if (range(sz[0] + 1, M1, M2)) b = 1;
 55    else break;
 56    --sz[b], ++sz[!b];
 57    sumGain.push_back(BL.mx(b));
 58    int node = BL.pop(b);
 59    if (node < 0) break;
 60    Lock[node] = true;
 61    swapNode.push_back(node);
 62    for (int i : Cells[node]) {
 63      --CntCell[b][i], ++CntCell[!b][i];
 64      for (int v : Nets[i]) {
 65        if (Lock[v]) continue;
 66        if (!CntCell[b][i] && Partition[v] != b) {
 67          BL.remove(v);
 68          BL.add(v, !b, --Gain[v]);
 69        } else if (CntCell[b][i] == 1 && Partition[v] == b) {
 70          BL.remove(v);
 71          BL.add(v, b, ++Gain[v]);
 72        }
 73      }
 74    }
 75  }
 76  int mx_i = 0;
 77  for (int i = 1; i < sumGain.size(); ++i) {
 78    sumGain[i] += sumGain[i - 1];
 79    if (sumGain[i] > sumGain[mx_i]) mx_i = i;
 80  }
 81  if (sumGain.size() && sumGain[mx_i] > 0)
 82    for (int i = 0; i <= mx_i; ++i)
 83      Partition[swapNode[i]] = !Partition[swapNode[i]];
 84  int ret = 0;
 85  for (int i = 0; i < M; ++i) {
 86    bool cut = false;
 87    for (int v : Nets[i])
 88      if (Partition[v] != Partition[Nets[i][0]]) cut = true;
 89    ret += cut;
 90  }
 91  return ret;
 92}

 94inline void genRandomBit(vector<bool> &v, unsigned int &seed) {
 95  int sz = v.size();
 96  v.clear();
 97  v.resize(sz);
 98  fill(v.begin(), v.begin() + (sz >> 1), 1);
 99  auto random = std::default_random_engine(seed);
100  shuffle(v.begin(), v.end(), random);
101  seed = random();
102}

104  auto StartTime = std::chrono::system_clock::now();
105  freopen(argc[1], "r", stdin);
106  freopen("output.txt", "w", stdout);
107  ios::sync_with_stdio(0), cin.tie(0);

112    string s;
113    getline(cin, s);
114    stringstream ss;
115    ss.str(s);

118    P += Nets[i].size();
119  }
120  int run = 16;
121  srand(112098);
122  vector<unsigned int> Seeds(run);
123  for (int i = 0; i < run; i++) Seeds[i] = rand();
124  vector<vector<bool>> Partition(run, vector<bool>(N));

126#pragma omp parallel for
127  for (int i = 0; i < run; ++i) {
128    vector<bool> tmp(N);
129    genRandomBit(tmp, Seeds[i]);
130    deque<int> buffer;
131    int buffer_avg = 0;
132    const int buffer_size = 10;
133    while (1) {
134      int cut = FM(tmp, P);
135      buffer.emplace_back(cut);
136      buffer_avg += cut;
137      if (buffer.size() > buffer_size) {
138        buffer_avg -= buffer.front();
139        buffer.pop_front();
140      }
141      if (buffer.size() == buffer_size &&
142          cut * buffer.size() >= buffer_avg)
143        genRandomBit(tmp, Seeds[i]);
144      else if (cut < CntCut[i])
145        CntCut[i] = cut, Partition[i] = tmp;
146      auto CurrentTime = std::chrono::system_clock::now();
147      if (CurrentTime - StartTime > std::chrono::seconds(80)) break;
148    }
149  }
150  int ans = 0;
151  for (int i = 1; i < run; ++i)
152    if (CntCut[i] < CntCut[ans]) ans = i;
153  for (int i = 0; i < N; ++i)
154    cout << Partition[ans][i] << '\n';
155  return 0;
156}
