  4  map<int, unordered_set<int>> mp[2];

  7public:
  8  int mx(int b) {
  9    while (mp[b].size() && mp[b].rbegin()->second.empty())
 10      mp[b].erase(mp[b].rbegin()->first);
 11    if (mp[b].size() == 0) return -1;
 12    return mp[b].rbegin()->first;
 13  }
 14  int pop(int b) {
 15    if (mx(b) < 0) return -1;
 16    int ret = *(mp[b].at(mx(b)).begin());
 17    mp[b].at(mx(b)).erase(ret);
 18    return ret;
 19  }

 24  }

 29int FM(vector<bool> &Partition, const int P) {
 30  int N = Cells.size(), M = Nets.size();
 31  int M1 = ceil(N * 0.451), M2 = floor(N * 0.549);
 32  vector<bool> Lock(N);
 33  vector<int> Gain(N), CntCell[2];
 34  CntCell[0].resize(M), CntCell[1].resize(M);
 35  for (int i = 0; i < M; i++) {
 36    for (int j : Nets[i])
 37      CntCell[Partition[j]][i]++;
 38    for (int j : Nets[i])
 39      if (!CntCell[!Partition[j]][i]) --Gain[j];
 40      else if (CntCell[Partition[j]][i] == 1) ++Gain[j];
 41  }

 44    BL.add(i, Partition[i], Gain[i]);
 45  vector<int> sumGain;
 46  vector<int> swapNode;
 47  int sz[2] = {};
 48  for (auto i : Partition) sz[i]++;
 49  for (int t = 0; t < N; ++t) {
 50    bool b;
 51    if (range(sz[0] - 1, M1, M2) && BL.mx(0) >= BL.mx(1)) b = 0;
 52    else if (range(sz[0] + 1, M1, M2) && BL.mx(0) <= BL.mx(1)) b = 1;
 53    else if (range(sz[0] - 1, M1, M2)) b = 0;
 54    else if (range(sz[0] + 1, M1, M2)) b = 1;
 55    else break;
 56    --sz[b], ++sz[!b];
 57    sumGain.push_back(BL.mx(b));
 58    int node = BL.pop(b);
 59    if (node < 0) break;
 60    Lock[node] = true;
 61    swapNode.push_back(node);
 62    for (int i : Cells[node]) {
 63      --CntCell[b][i], ++CntCell[!b][i];
 64      for (int v : Nets[i]) {
 65        if (Lock[v]) continue;
 66        if (!CntCell[b][i] && Partition[v] != b) {
 67          BL.remove(v);
 68          BL.add(v, !b, --Gain[v]);
 69        } else if (CntCell[b][i] == 1 && Partition[v] == b) {

 72        }
 73      }
 74    }

 77  for (int i = 1; i < sumGain.size(); ++i) {
 78    sumGain[i] += sumGain[i - 1];
 79    if (sumGain[i] > sumGain[mx_i]) mx_i = i;
 80  }
 81  if (sumGain.size() && sumGain[mx_i] > 0)
 82    for (int i = 0; i <= mx_i; ++i)

 88      if (Partition[v] != Partition[Nets[i][0]]) cut = true;
 89    ret += cut;
 90  }
