3.3 Home on the Range

  • 題意：給定一個邊長為 n ≤ 250 的 0/1 正方形，問邊長大於 2 的各種大小正方形數量？
  • 解法：Prefix sum，O(n^3)。
 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    freopen("range.in", "r", stdin);
 5    freopen("range.out", "w", stdout);
 6    ios::sync_with_stdio(0), cin.tie(0);
 7    int n; cin >> n;
 8    vector<string> v(n);
 9    for (int i = 0; i < n; i++) cin >> v[i];
10    vector<vector<int>> cnt(n, vector<int>(n));
11    for (int i = 0; i < n; i++)
12        for (int j = 0; j < n; j++)
13            if (v[i][j] == '1')
14                cnt[i][j] = (j ? cnt[i][j - 1] : 0) + 1;
15    vector<int> ans(n + 1);
16    for (int i = 0; i < n; i++) {
17        for (int j = 0; j < n; j++) {
18            for (int k = 0, mi = n; i + k < n; k++) {
19                mi = min(mi, cnt[i + k][j]);
20                if (mi > k) ans[k + 1]++;
21                else break;
22            }
23        }
24    }
25    for (int i = 2; i <= n && ans[i]; i++)
26        cout << i << " " << ans[i] << "\n";
27    return 0;
28}

3.3 A Game

  • 題意：給定一個長度為 n ≤ 100 的整數陣列，元素介於 1 ~ 200。兩個玩家輪流從頭或尾拿一個元素，假設兩人都用最佳策略，問最後得到的元素和？
  • 解法：DP，O(n^2)。
 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    freopen("game1.in", "r", stdin);
 5    freopen("game1.out", "w", stdout);
 6    int n; cin >> n;
 7    vector<int> a(n);
 8    int sum = 0;
 9    for (int i = 0; i < n; i++) {
10        cin >> a[i];
11        sum += a[i];
12    }
13    vector<vector<int>> dp(n, vector<int>(n));
14    for (int i = 0; i < n; i++)
15        dp[i][i] = a[i];
16    for (int i = 1; i < n; i++)
17        for (int l = 0, r = i; r < n; l++, r++)
18            if (i % 2) dp[l][r] = min(-a[l] + dp[l + 1][r], -a[r] + dp[l][r - 1]);
19            else dp[l][r] = max(a[l] + dp[l + 1][r], a[r] + dp[l][r - 1]);
20                
21    int x = (sum + dp[0][n - 1]) / 2;
22    int y = sum - x;
23    if (n % 2) cout << x << " " << y << "\n";
24    else cout << y << " " << x << "\n";
25    return 0;
26}
