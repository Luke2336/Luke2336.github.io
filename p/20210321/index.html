GCC 版本問題

CUDA 只支援 GCC5 以前的版本，幸好工作站上有 GCC4.8，所以用 nvcc --compiler-binder /usr/bin/gcc4.8 就解決了編譯的問題了。

Note

  • Host vs. Device
    • Host 是用 CPU 跑 serial 的部分。
    • Device 是用 GPU 跑 parallel 的部分。
    • 兩者的 Memory 是分開的。
  • __global__ 要加在 device function code 前面。
  • cudaMalloc()、cudaMemcpy()、cudaFree()。

Simple Code

 1#include<stdio.h>
 2using namespace std;
 3__global__ void add(int *a, int *b, int *c) {
 4    *c = *a + *b;
 5}
 6int main() {
 7    int a, b, c;
 8    int *d_a, *d_b, *d_c;
 9    int size = sizeof(int);
10    cudaMalloc((void**)&d_a, size);
11    cudaMalloc((void**)&d_b, size);
12    cudaMalloc((void**)&d_c, size);
13    a = 2, b = 7;
14    cudaMemcpy(d_a, &a, size, cudaMemcpyHostToDevice);
15    cudaMemcpy(d_b, &b, size, cudaMemcpyHostToDevice);
16    printf("a = %d\n", a);
17    printf("b = %d\n", b);
18    add<<<1, 1>>>(d_a, d_b, d_c);
19    cudaMemcpy(&c, d_c, size, cudaMemcpyDeviceToHost);
20    cudaFree(d_a);
21    cudaFree(d_b);
22    cudaFree(d_c);
23    printf("c = %d\n", c);
24    return 0;
25}
Vector 加法
  • 改成 add<<<1, N>>>(d_a, d_b, d_c); 即可用 threadIdx.x 操作
 1#include<stdio.h>
 2using namespace std;
 3__global__ void add(int *a, int *b, int *c) {
 4    c[blockIdx.x] = a[blockIdx.x] + b[blockIdx.x];
 5}
 6#define N 20
 7int main() {
 8    int *a, *b, *c;
 9    int *d_a, *d_b, *d_c;
10    int size = N * sizeof(int);
11    cudaMalloc((void**)&d_a, size);
12    cudaMalloc((void**)&d_b, size);
13    cudaMalloc((void**)&d_c, size);
14    a = (int*)malloc(size);
15    b = (int*)malloc(size);
16    c = (int*)malloc(size);
17    for (int i = 0; i < N; i++)
18        a[i] = b[i] = i;
19    cudaMemcpy(d_a, a, size, cudaMemcpyHostToDevice);
20    cudaMemcpy(d_b, b, size, cudaMemcpyHostToDevice);
21    add<<<N, 1>>>(d_a, d_b, d_c);
22    cudaMemcpy(c, d_c, size, cudaMemcpyDeviceToHost);
23    cudaFree(d_a); cudaFree(d_b); cudaFree(d_c);
24    free(a); free(b); free(c);
25    for (int i = 0; i < N; i++)
26        printf("%d %d %d\n", a[i], b[i], c[i]);
27    return 0;
28}
  • 多個 block，每個 block 用 M 個 Thread。
  • 每個 Block 會輪流執行，由多個 Thread 去跑完一個 Block 才跑下一個，同一個 Block 中的 Thread 可 shared memory。
 1#include<vector>
 2#include<stdio.h>
 3using namespace std;
 4#define N 1000
 5#define M 30
 6__global__ void add(int *a, int *b, int *c) {
 7    int index = threadIdx.x + blockIdx.x * blockDim.x;
 8    if (index < N)
 9        c[index] = a[index] + b[index];
10}
11int main() {
12    int *a, *b, *c;
13    int *d_a, *d_b, *d_c;
14    int size = N * sizeof(int);
15    cudaMalloc((void**)&d_a, size);
16    cudaMalloc((void**)&d_b, size);
17    cudaMalloc((void**)&d_c, size);
18    a = (int*)malloc(size);
19    b = (int*)malloc(size);
20    c = (int*)malloc(size);
21    for (int i = 0; i < N; i++)
22        a[i] = b[i] = i;
23    cudaMemcpy(d_a, a, size, cudaMemcpyHostToDevice);
24    cudaMemcpy(d_b, b, size, cudaMemcpyHostToDevice);
25    add<<<(N+M-1)/M, M>>>(d_a, d_b, d_c);
26    cudaMemcpy(c, d_c, size, cudaMemcpyDeviceToHost);
27    free(a); free(b); free(c);
28    cudaFree(d_a); cudaFree(d_b); cudaFree(d_c);
29    return 0;
30}
