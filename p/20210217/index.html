 5int C(int a, int b) {
 6    if (a < b) return 0;
 7    vector<int> up, down;
 8    for (int i = a; i > b; i--) up.push_back(i);
 9    for (int i = a - b; i > 1; i--) down.push_back(i);

15    }
16    int ret = 1;
17    for (int x : up) ret *= x;
18    return ret;
19}
20void solve(int len, int cnt, ui id) {
