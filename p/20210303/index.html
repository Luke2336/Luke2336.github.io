pC

只要是在 k 個點中任意兩個的路徑上就可以選。

 1#include<bits/stdc++.h>
 2using namespace std;
 3vector<vector<int>> G;
 4vector<bool> color;
 5int k;
 6int ans = 0;
 7int dfs(int u, int p) {
 8    int cnt = 0, r = 0;
 9    for (int uu : G[u]) {
10        if (uu == p) continue;
11        int tmp = dfs(uu, u);
12        if (tmp) r++, cnt += tmp;
13    }
14    cnt += color[u];
15    if (color[u]) ans++;
16    else {
17        if (k - cnt) r++;
18        if (r >= 2) ans++;
19    }
20    return cnt;
21}
22int main () {
23    ios::sync_with_stdio(0), cin.tie(0);
24    int n;
25    cin >> n >> k;
26    G.resize(n + 1);
27    color.resize(n + 1);
28    for (int i = 1; i < n; i++) {
29        int a, b, w;
30        cin >> a >> b >> w;
31        G[a].push_back(b);
32        G[b].push_back(a);
33    }
34    for (int i = 0; i < k ; i++) {
35        int c; cin >> c;
36        color[c] = true;
37    }
38    dfs(1, 0);
39    cout << ans << "\n";
40    return 0;
41}

pH

用 FFT 對上下兩排做多項式乘法，再用中間那排算係數和。

 1#include<bits/stdc++.h>
 2using namespace std;
 3typedef long long ll;
 4template <typename T, typename VT = vector<complex<T>> >
 5struct FFT {
 6    const T pi = acos((T)-1.0);
 7    unsigned bit_reverse(unsigned a, int len) {
 8        a = ((a&0x55555555U)<<1) | ((a&0xAAAAAAAAU)>>1);
 9        a = ((a&0x33333333U)<<2) | ((a&0xCCCCCCCCU)>>2);
10        a = ((a&0x0F0F0F0FU)<<4) | ((a&0xF0F0F0F0U)>>4);
11        a = ((a&0x00FF00FFU)<<8) | ((a&0xFF00FF00U)>>8);
12        a = ((a&0x0000FFFFU)<<16) | ((a&0xFFFF0000U)>>16);
13        return a >> (32 - len);
14    }
15    void fft(bool is_inv, VT &in, VT &out, int N) {
16        int bitlen = __lg(N), num = is_inv ? 1 : -1;
17        for (int i = 0; i < N; ++i)
18            out[bit_reverse(i, bitlen)] = in[i];
19        for (int step = 2, mh = 1; step <= N; step <<= 1, mh <<= 1) {
20            for (int i = 0; i < mh; ++i) {
21                complex<T> wi = exp(complex<T>(0, i * num * pi / mh));
22                for (int j = i, k = i + mh; j < N; j += step, k += step) {
23                    complex<T> u = out[j], t = wi * out[k];
24                    out[j] = u + t, out[k] = u - t;
25                }
26            }
27        }
28        for (int i = 0; is_inv && i < N; ++i)
29            out[i] /= N;
30    }
31};
32ll solve(vector<vector<int>> a, int *n) {
33    int mx = 0, N = 131072;
34    FFT<double> F;
35    vector<complex<double>> v[3];
36    vector<complex<double>> a_fft(N), b_fft(N), ab_fft(N), ab(N);
37    for (int i = 0; i < 3; i++) {
38        v[i].resize(N, complex<double>(0));
39        for (int j = 0; j < n[i]; j++)
40            v[i][a[i][j] + 30000] = complex<double>(1);
41    }
42    F.fft(0, v[0], a_fft, N);
43    F.fft(0, v[2], b_fft, N);
44    for (int i = 0; i < N; i++)
45        ab_fft[i] = a_fft[i] * b_fft[i];
46    F.fft(1, ab_fft, ab, N);
47    ll ans = 0;
48    for (int i = 0; i < n[1]; i++) {
49        int goal = (a[1][i] + 30000) << 1;
50        ans += ll(ab[goal].real() + 1e-6);
51    }
52    return ans;
53}
54int main () {
55    int n[3];
56    vector<vector<int>> a(3);
57    for (int i = 0 ; i < 3; i++) {
58        cin >> n[i];
59        a[i].resize(n[i]);
60        for (int j = 0; j < n[i]; j++)
61            cin >> a[i][j];
62    }
63    cout << solve(a, n) << "\n";
64    return 0;
65}
