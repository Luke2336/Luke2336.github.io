 6#include <bits/stdc++.h>
 7using namespace std;
 8typedef pair<int, int> pii;
 9#define x first
10#define y second
11#define INF 100000000
12inline int step(pii k, pii p) {
13    return max(abs(k.x - p.x), abs(k.y - p.y));
14}
15inline bool range(pii p, int R, int C) {
16    return 0 <= p.x && 0 <= p.y && p.x < R && p.y < C;
17}
18int dis[1000][30][35][2];
19int main() {
20    freopen("camelot.in", "r", stdin);
21    freopen("camelot.out", "w", stdout);
22    int R, C;
23    cin >> R >> C;
24    int tmp_int;
25    char tmp_ch;
26    cin >> tmp_ch >> tmp_int;
27    pii king = {tmp_int - 1, tmp_ch - 'A'};
28    vector<pii> knight;
29    while (cin >> tmp_ch >> tmp_int)
30        knight.push_back({tmp_int - 1, tmp_ch - 'A'});
31    if (!knight.size()) {
32        cout << "0\n";
33        return 0;
34    }
35    for (int i = 0; i < knight.size(); i++)
36        for (int j = 0; j < R; j++)
37            for (int k = 0; k < C; k++)
38                dis[i][j][k][0] = dis[i][j][k][1] = INF;
39    const int dx[8] = {2, 2, 1, 1, -1, -1, -2, -2};
40    const int dy[8] = {1, -1, 2, -2, 2, -2, 1, -1};
41    for (int i = 0; i < knight.size(); i++) {
42        dis[i][knight[i].x][knight[i].y][0] = 0;
43        dis[i][knight[i].x][knight[i].y][1] = step(king, knight[i]);
44        queue<pii> q;
45        q.push(knight[i]);
46        while (!q.empty()) {
47            int xx = q.front().x, yy = q.front().y;
48            q.pop();
49            for (int di = 0; di < 8; di++) {
50                int a = xx + dx[di], b = yy + dy[di];
51                if (!range({a, b}, R, C)) continue;
52                bool inque = false;
53                if (dis[i][a][b][0] > dis[i][xx][yy][0] + 1) {
54                    dis[i][a][b][0] = dis[i][xx][yy][0] + 1;
55                    inque = true;
56                }
57                int d = min(dis[i][xx][yy][0] + step(king, {a, b}),
58                            dis[i][xx][yy][1]) + 1;
59                if (dis[i][a][b][1] > d) {
60                    dis[i][a][b][1] = d;
61                    inque = true;
62                }
63                if (inque) q.push({a, b});
64            }
65        }
66    }
67    int ans = INF;
68    for (int i = 0; i < R; i++) {
69        for (int j = 0; j < C; j++) {
70            int mi = INF, sum = 0;
71            for (int k = 0; k < knight.size(); k++) {
72                sum += dis[k][i][j][0];
73                mi = min(mi, dis[k][i][j][1] - dis[k][i][j][0]);
74            }
75            ans = min(ans, sum + mi);
76        }
77    }
