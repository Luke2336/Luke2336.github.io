pB. Gremlins attack!

  • 題意：有 m 個人，及依序關 k 盞燈。人只能走在有關燈的地方，問最少要關幾盞燈時，至少有一個人能走到邊界？
  • 解法：BFS＋priority_queue，答案為走到邊界的路徑上，最大編號的燈。
 1#pragma GCC optimization ("O3")
 2#include<bits/stdc++.h>
 3using namespace std;
 4typedef pair<int, int> pii;
 5typedef pair<int, pii> piii;
 6#define pb push_back
 7int dx[4] = {1, -1, 0, 0};
 8int dy[4] = {0, 0, 1, -1};
 9int main() {
10    ios::sync_with_stdio(0), cin.tie(0);
11    int n, m, k; cin >> n >> m >> k;
12    int a[505][505], dis[505][505];
13    memset(a, -1, sizeof(a)), memset(dis, -1, sizeof(dis));
14    priority_queue<piii, vector<piii>, greater<piii> > pq;
15    for (int i = 0; i < m; i++) {
16        int x, y; cin >> x >> y;
17        a[x][y] = dis[x][y] = 0;
18        pq.push(piii(0, pii(x, y)));
19    }
20    for (int i = 1; i <= k; i++) {
21        int x, y; cin >> x >> y;
22        a[x][y] = i;
23    }
24    while (!pq.empty()) {
25        int x = pq.top().second.first;
26        int y = pq.top().second.second;
27        pq.pop();
28        if (x == 0 || y == 0 || x == n - 1 || y == n - 1) {
29            cout << dis[x][y] << "\n";
30            return 0;
31        }
32        for (int d = 0; d < 4; d++) {
33            int xx = x + dx[d], yy = y + dy[d];
34            if (xx < 0 || xx >= n || yy < 0 || yy >= n) continue;
35            if (a[xx][yy] == -1) continue;
36            int tmp = max(a[xx][yy], dis[x][y]);
37            if (dis[xx][yy] == -1 || tmp < dis[xx][yy])
38                dis[xx][yy] = tmp,
39                pq.push(piii(dis[xx][yy], pii(xx, yy)));
40        }
41    }
42    return 0;
43}

pG

  • 題意：給定一個 n 以及一個布林運算式，此運算式會定義連續十個以內的位元的函數，求有多少個長度為 n 的 01 字串使得任意子字串都符合此運算式。
  • 解法：枚舉長度及最後十位的狀態，因為每個狀態接下來只會補 0 或 1，所以轉移是 O(1)，狀態共 O(2^{10} n)。打表預處理每個狀態是否合法。
 1#include <bits/stdc++.h>
 2#define int ll
 3using namespace std;
 4#define vt          vector
 5#define sz(x)       int((x).size())
 6#define all(x)      (x).begin(), (x).end()
 7#define rall(x)     (x).rbegin(), (x).rend()
 8#define fi          first
 9#define se          second
10using ll = long long;
11using pii = pair<int, int>;
12const ll MOD = (ll)1 << 60;
13int n, p;
14string s;
15int eval(int x) {
16    int neg = 0, val = 1;
17    char op = 0;
18    while (p < sz(s)) {
19        if (isdigit(s[p])) {
20            int next = ((x >> (s[p] - '0')) ^ neg) & 1;
21            neg = 0;
22            if (op == '&') val &= next;
23            else if (op == '|') val |= next;
24            else if (op == '+') val ^= next;
25            else val = next;
26            p++;
27        } else if (s[p] == '(') {
28            p++;
29            int next = (eval(x) ^ neg) & 1;
30            neg = 0;
31            if (op == '&') val &= next;
32            else if (op == '|') val |= next;
33            else if (op == '+') val ^= next;
34            else val = next;
35        } else if (s[p] == ')') {
36            p++;
37            return val;
38        } else if (s[p] == '-') {
39            p++, neg ^= 1;
40        } else {
41            op = s[p++];
42        }
43    }
44    return val;
45}
46void solve() {
47    cin >> n >> s;
48    int k = 0;
49    for(char i = '9', flag = 0; i >= '0'; i--){
50        for(char j: s){
51            if(i == j){
52                k = i - '0';
53                flag = 1;
54                break;
55            }
56        }
57        if(flag) break;
58    }
59    vector<int> tb(2 << k);
60    for (int i = 0; i < (2 << k); i++) {
61        p = 0;
62        tb[i] = eval(i);
63    }
64    int rolling = 1;
65    vector<vector<int>> dp(2, vector<int>(1 << (k + 1)));
66    for(int i = 0; i < (2 << k); i++)
67        dp[not rolling][i] = tb[i];
68    for(int i = 0; i < n - k - 1; i++){
69        for(int j = 0; j < (2 << k); j++)
70            dp[rolling][j] = 0;
71        for(int j = 0; j < (2 << k); j++){
72            int r = (j << 1) % (1 << (k + 1));
73            dp[rolling][r] += dp[not rolling][j] * tb[r];
74            dp[rolling][r + 1] += dp[not rolling][j] * tb[r + 1];
75            dp[rolling][r] %= MOD;
76            dp[rolling][r + 1] %= MOD;
77        }
78        rolling ^= 1;
79    }
80    int output = 0;
81    for(int i = 0; i < (2 << k); i++){
82        output += dp[not rolling][i];
83        output %= MOD;
84    }
85    cout << output;
86}
87int32_t main() {
88    ios::sync_with_stdio(0), cin.tie(0);
89    solve();
90}
