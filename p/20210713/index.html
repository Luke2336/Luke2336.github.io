SPOJ - AGGRCOW

 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    int T;
 5    cin >> T;
 6    for (int t = 0; t < T; ++t) {
 7        int n, c;
 8        cin >> n >> c;
 9        vector<int> a(n);
10        for (int i = 0; i < n; ++i)
11            cin >> a[i];
12        sort(a.begin(), a.end());
13        int L = 1, R = a.back();
14        int ans = 0;
15        while (L <= R) {
16            int mid = (L + R) >> 1;
17            int last = 0, now = 0;
18            bool success = true;
19            for (int i = 1; i < c; i++) {
20                while (a[now] - a[last] < mid && now < n)
21                    ++now;
22                last = now;
23                if (now >= n) {
24                    success = false;
25                    break;
26                }
27            }
28            if (success) L = mid + 1, ans = max(ans, mid);
29            else R = mid - 1;
30        }
31        cout << ans << "\n";
32    }
33    return 0;
34}

SPOJ - CANDY

 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    while (1) {
 5        int n;
 6        cin >> n;
 7        if (n == -1) break;
 8        vector<int> v(n);
 9        int sum = 0;
10        for (int i = 0; i < n; ++i) {
11            cin >> v[i];
12            sum += v[i];
13        }
14        if (sum % n)
15            cout << "-1\n";
16        else {
17            int ans = 0;
18            for (int i = 0; i < n; i++)
19                ans += max(0, v[i] - sum / n);
20            cout << ans << "\n";
21        }
22    }
23    return 0;
24}
